    const progCtx = document.getElementById('programChart');

            data: {

                    borderWidth: 2,
                    borderColor: '#fff',
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: {
                        position: 'bottom',

                    }
                }
            }
        });
